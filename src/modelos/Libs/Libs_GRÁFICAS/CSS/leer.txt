import cssutils

# Example of parsing a CSS string
css = """
body {
    background-color: #f0f0f0;
    color: #333;
}
"""
stylesheet = cssutils.parseString(css)

# Accessing rules
for rule in stylesheet:
    print(rule.cssText)
